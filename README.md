def main(z, x, y):
    n = len(x)
    s = 0
    x.insert(0, 0)
    y.insert(0, 0)
    z.insert(0, 0)

        s += (3 * z[math.ceil(i / 4)] ** 2 + y[math.ceil(i / 2)] +

def main(x):
    a = 0x7f
    b = 0x7f << 7
    c = 0xf << 14
    d = 0x7ff << 18
    e = 0x7 << 29

    e = (e & x) >> 29
    d = (d & x) >> 4
    c = (c & x) >> 11
    b = (b & x) << 18
    a = (a & x) << 7
    return c | b | a | d | e
